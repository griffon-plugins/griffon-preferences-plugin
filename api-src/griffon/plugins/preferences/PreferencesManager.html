16 package griffon.plugins.preferences;
17 
18 import javax.annotation.Nonnull;
19 
20 /**
21  * @author Andres Almiray
22  */
23 public interface PreferencesManager {
24     /**
25      * Returns the {@code Preferences} instance contained by this manager.</p>
26      * Never returns null.
27      *
28      * @return the {@code Preferences} instance contained by this manager.
29      */
30     @Nonnull
31     Preferences getPreferences();
32 
33     /**
34      * Loads a matching {@code PreferencesNode} into the target instance.
35      *
36      * @param instance an object with fields/properties annotated with {@code @Preference}.
37      */
38     void injectPreferences(@Nonnull Object instance);
39 
40     /**
41      * <p>Merges all fields/properties annotated with {@code @Preference} to their
42      * corresponding {@code PreferencesNode} within the contained {@code Preferences}.</p>
43      * <p>Values will be transformed their literal representation if a <code>format</code>
44      * is present in their {@code @Preference} annotation. Null values will cause
45      * the removal of the key in the {@code PreferencesNode}.</p>
46      *
47      * @param instance an object with fields/properties annotated with {@code @Preference}.
48      */
49     void save(@Nonnull Object instance);
50 }
